### Pub/Sub System

```javascript
// Publisher
const publisher = client.pubsub();
await publisher.publish("events", {
  type: "user.signup",
  userId: "123",
  timestamp: Date.now(),
});

// Subscriber
const subscriber = client.pubsub();
await subscriber.subscribe("events", (message) => {
  console.log("Received:", message);
});
```

## üíæ Data Features

### Advanced Querying

```javascript
// Complex queries with filtering, sorting, pagination
const results = await client.data.query({
  collection: "users",
  filter: {
    status: "active",
    age: { $gte: 18 },
    tags: { $in: ["premium", "verified"] },
  },
  sort: { createdAt: -1 },
  limit: 100,
  offset: 0,
});
```

### Batch Operations

```javascript
// Bulk insert
const users = await client.users.bulkCreate([
  { name: "User 1", email: "user1@example.com" },
  { name: "User 2", email: "user2@example.com" },
  // ... up to 10,000 records
]);

// Bulk update
await client.users.bulkUpdate({
  filter: { status: "pending" },
  update: { status: "active" },
});
```

### Data Streaming

```javascript
// Stream large datasets efficiently
const stream = client.data.stream("users", {
  batchSize: 1000,
  filter: { status: "active" },
});

for await (const batch of stream) {
  // Process 1000 records at a time
  await processBatch(batch);
}
```
