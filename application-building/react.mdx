## Creating a Component

```typescript
import React, { useState } from "react";
import { registerComponent, Prop } from "@superblocks/library";

const properties = {
  title: Prop.string().default("My Component"),
  count: Prop.number().default(0),
  onClick: Prop.event(),
};

const MyComponent = ({ title, count, onClick }) => {
  const [localCount, setLocalCount] = useState(count);

  const handleClick = () => {
    setLocalCount(localCount + 1);
    onClick?.();
  };

  return (
    <div style={{ padding: "20px" }}>
      <h2>{title}</h2>
      <p>Count: {localCount}</p>
      <button onClick={handleClick}>Increment</button>
    </div>
  );
};

export default registerComponent("MyComponent", properties, MyComponent);
```

## Example: Data Table Component

```typescript
import React from "react";
import { registerComponent, Prop } from "@superblocks/library";

const properties = {
  data: Prop.array().default([]),
  columns: Prop.array().default([]),
  onRowClick: Prop.event(),
};

const DataTable = ({ data, columns, onRowClick }) => {
  return (
    <table style={{ width: "100%", borderCollapse: "collapse" }}>
      <thead>
        <tr>
          {columns.map((col, i) => (
            <th
              key={i}
              style={{
                padding: "12px",
                borderBottom: "2px solid #e5e7eb",
                textAlign: "left",
                fontWeight: 600,
              }}
            >
              {col.label}
            </th>
          ))}
        </tr>
      </thead>
      <tbody>
        {data.map((row, i) => (
          <tr
            key={i}
            onClick={() => onRowClick?.(row)}
            style={{
              cursor: "pointer",
              ":hover": { backgroundColor: "#f9fafb" },
            }}
          >
            {columns.map((col, j) => (
              <td
                key={j}
                style={{
                  padding: "12px",
                  borderBottom: "1px solid #e5e7eb",
                }}
              >
                {row[col.field]}
              </td>
            ))}
          </tr>
        ))}
      </tbody>
    </table>
  );
};

export default registerComponent("DataTable", properties, DataTable);
```
